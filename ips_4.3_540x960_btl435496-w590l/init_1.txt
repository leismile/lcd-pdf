void Spi_NT35580_Write(U16 Spi_Comm,U8 Spi_Data)
{
	U16	SBit,Write_Deta;
	U8 BitCounter;

	Write_Deta=(0x2000|((Spi_Comm>>8)&0x00FF));
	Set_CS(0);
	for(BitCounter=0;BitCounter<16;BitCounter++)
		{
			SBit = Write_Deta&0x8000;
			Set_SCK(0);			//CLR SCL
			if(SBit)
				Set_SDA(1);
			else
				Set_SDA(0);
			Delay(2);
			Set_SCK(1);			//SET SCL
			Delay(2);
			Write_Deta = Write_Deta<<1;
		}
	Set_CS(1);
	Write_Deta=(Spi_Comm&0x00FF);
	Set_CS(0);
	for(BitCounter=0;BitCounter<16;BitCounter++)
		{
			SBit = Write_Deta&0x8000;
			Set_SCK(0);			//CLR SCL
			if(SBit)
				Set_SDA(1);
			else
				Set_SDA(0);
			Delay(2);
			Set_SCK(1);			//SET SCL
			Delay(2);
			Write_Deta = Write_Deta<<1;
		}
	Set_CS(1);
	Write_Deta=(0x4000|Spi_Data);
	Set_CS(0);
	for(BitCounter=0;BitCounter<16;BitCounter++)
		{
			SBit = Write_Deta&0x8000;
			Set_SCK(0);			//CLR SCL
			if(SBit)
				Set_SDA(1);
			else
				Set_SDA(0);
			Delay(2);
			Set_SCK(1);			//SET SCL
			Delay(2);
			Write_Deta = Write_Deta<<1;
		}
	Set_CS(1);
}
